\newcommand{\harmonic}[6]{
	\foreach \x in {1,...,#4} {
		\pgfmathparse{Mod(\x,2)==0?0:1}
		\ifnum\pgfmathresult>0
        	\draw[#6] ({#1+(\x-1)*#3/#4},0) sin ({#1+(\x-0.5)*#3/#4},#5) cos ({#1+\x*#3/#4},0);      
    	\else
        	\draw[#6] ({#1+(\x-1)*#3/#4},0) sin ({#1+(\x-0.5)*#3/#4},-#5) cos ({#1+\x*#3/#4},0);                 
    	\fi
    }
}

\begin{tikzpicture}[scale=0.2]
\foreach \y/\z in {1/blue!100,3/blue!90,5/blue!80,7/blue!70,9/blue!60,11/blue!50,13/blue!40,15/blue!30} \harmonic{0}{0}{8}{\y}{\y/4}{\z};
\end{tikzpicture}
